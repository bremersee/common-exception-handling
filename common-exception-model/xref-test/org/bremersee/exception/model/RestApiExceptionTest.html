54      RestApiException model = new RestApiException();
55      model.setId("value");
56      softly.assertThat(model.getId()).isEqualTo("value");
57  
58      model = RestApiException.builder().id("value").build();
59      softly.assertThat(model.getId()).isEqualTo("value");

62      softly.assertThat(model).isNotEqualTo(new Object());
63      softly.assertThat(model).isEqualTo(model);
64      softly.assertThat(model).isEqualTo(model.toBuilder().id("value").build());
65  
66      softly.assertThat(model.toString()).contains("value");
67    }

343   /**
344    * Json.
345    *
346    * @param softly the soft assertions
347    * @throws Exception the exception
348    */
349   @Test
350   void json(SoftAssertions softly) throws Exception {
351     RestApiException cause = RestApiException.builder()
352         .id(UUID.randomUUID().toString())
353         .application("test")
354         .path("/api/somewhere")
355         .message("value")
356         .errorCode("1234")
357         .errorCodeInherited(Boolean.FALSE)
358         .timestamp(OffsetDateTime.parse("2007-12-24T18:20Z", ISO_OFFSET_DATE_TIME))
359         .build();
360     Handler handler = Handler.builder()
361         .methodName("getSomething")
362         .methodParameterTypes(Arrays.asList("java.lang.String", "java.lang.Boolean"))
363         .className("org.bremersee.SomethingController")
364         .build();
365     StackTraceItem i0 = StackTraceItem.builder()
366         .methodName("getSomething")
367         .lineNumber(123)
368         .declaringClass("org.bremersee.SomethingController")
369         .build();
370     StackTraceItem i1 = StackTraceItem.builder()
371         .methodName("findSomething")
372         .lineNumber(456)
373         .declaringClass("org.bremersee.SomethingRepository")
374         .build();
375     RestApiException model = RestApiException.builder()
376         .cause(cause)
377         .handler(handler)
378         .stackTrace(Arrays.asList(i0, i1))
379         .errorCodeInherited(true)
380         .errorCode(cause.getErrorCode())
381         .path("(api/something")
382         .id("5678")
383         .message("Something went wrong.")
384         .application("junit")
385         .timestamp(OffsetDateTime.parse("2007-12-24T18:21Z", ISO_OFFSET_DATE_TIME))
386         .build();
387 
388     ObjectMapper objectMapper = new ObjectMapper();
389     objectMapper.registerModule(new Jdk8Module());
390     objectMapper.registerModule(new JavaTimeModule());
391     objectMapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
392     objectMapper.enable(SerializationFeature.WRITE_DATES_WITH_ZONE_ID);
393 
394     String json = objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(model);
395     // System.out.println(json);
396 
397     RestApiException actualModel = objectMapper.readValue(json, RestApiException.class);
398     softly.assertThat(actualModel).isEqualTo(model);
399   }

407   @Test
408   void xml(SoftAssertions softly) throws Exception {
409     RestApiException cause = RestApiException.builder()
410         .id(UUID.randomUUID().toString())
411         .application("test")
412         .path("/api/somewhere")
413         .message("value")
414         .errorCode("1234")
415         .errorCodeInherited(Boolean.FALSE)
416         .timestamp(OffsetDateTime.parse("2007-12-24T18:20Z", ISO_OFFSET_DATE_TIME))
417         .build();
418     Handler handler = Handler.builder()
419         .methodName("getSomething")
420         .methodParameterTypes(Arrays.asList("java.lang.String", "java.lang.Boolean"))
421         .className("org.bremersee.SomethingController")
422         .build();
423     StackTraceItem i0 = StackTraceItem.builder()
424         .methodName("getSomething")
425         .lineNumber(123)
426         .declaringClass("org.bremersee.SomethingController")
427         .build();
428     StackTraceItem i1 = StackTraceItem.builder()
429         .methodName("findSomething")
430         .lineNumber(456)
431         .declaringClass("org.bremersee.SomethingRepository")
432         .build();
433     RestApiException model = RestApiException.builder()
434         .cause(cause)
435         .handler(handler)
436         .stackTrace(Arrays.asList(i0, i1))
437         .errorCodeInherited(true)
438         .errorCode(cause.getErrorCode())
439         .path("(api/something")
440         .id("5678")
441         .message("Something went wrong.")
442         .application("junit")
443         .timestamp(OffsetDateTime.parse("2007-12-24T18:21Z", ISO_OFFSET_DATE_TIME))
444         .build();
445     model.furtherDetails("custom", "A custom information");
446 
447     XmlMapper xmlMapper = new XmlMapper();
448     xmlMapper.registerModule(new Jdk8Module());
449     xmlMapper.registerModule(new JavaTimeModule());
450     xmlMapper.enable(Feature.WRITE_XML_DECLARATION);
451     xmlMapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
452     xmlMapper.enable(SerializationFeature.WRITE_DATES_WITH_ZONE_ID);
453     String xml = xmlMapper
454         .writerWithDefaultPrettyPrinter()
455         .writeValueAsString(model);
456     System.out.println("Plain jackson without jaxb module:");
457     System.out.println(xml);
458 
459     RestApiException actualModel = xmlMapper
460         .readValue(new StringReader(xml), RestApiException.class);
461     softly.assertThat(actualModel).isEqualTo(model);
462   }
