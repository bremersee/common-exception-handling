17  package org.bremersee.exception.spring.boot.autoconfigure.reactive;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.bremersee.exception.RestApiExceptionMapper;
21  import org.bremersee.exception.spring.boot.autoconfigure.RestApiExceptionMapperBootProperties;
22  import org.springframework.beans.factory.ObjectProvider;
23  import org.springframework.boot.autoconfigure.AutoConfiguration;
24  import org.springframework.boot.autoconfigure.AutoConfigureAfter;
25  import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
26  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
27  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
28  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication.Type;
29  import org.springframework.boot.autoconfigure.web.WebProperties;
30  import org.springframework.boot.context.event.ApplicationReadyEvent;
31  import org.springframework.boot.context.properties.EnableConfigurationProperties;
32  import org.springframework.boot.web.reactive.error.ErrorAttributes;
33  import org.springframework.context.ApplicationContext;
34  import org.springframework.context.annotation.Bean;
35  import org.springframework.context.event.EventListener;
36  import org.springframework.core.annotation.Order;
37  import org.springframework.http.codec.ServerCodecConfigurer;
38  import org.springframework.util.Assert;
39  import org.springframework.util.ClassUtils;
40  
41  /**
42   * The api exception handler autoconfiguration.
43   *
44   * @author Christian Bremer
45   */
46  @ConditionalOnWebApplication(type = Type.REACTIVE)
47  @ConditionalOnClass(name = {
48      "com.fasterxml.jackson.databind.ObjectMapper",
49      "org.bremersee.exception.RestApiExceptionMapper"
50  })
51  @ConditionalOnBean({
52      ErrorAttributes.class,
53      WebProperties.class,
54      ServerCodecConfigurer.class
55  })
56  @AutoConfigureAfter({
57      RestApiExceptionMapperForWebFluxAutoConfiguration.class
58  })
59  @AutoConfiguration
60  @EnableConfigurationProperties({RestApiExceptionMapperBootProperties.class})
61  @Slf4j
62  public class ApiExceptionHandlerAutoConfiguration {
63  
64    private final RestApiExceptionMapperBootProperties properties;
65  
66    /**
67     * Instantiates a new api exception handler autoconfiguration.
68     *
69     * @param properties the properties
70     */
71    public ApiExceptionHandlerAutoConfiguration(
72        RestApiExceptionMapperBootProperties properties) {
73      this.properties = properties;
74    }
75  
76    /**
77     * Init.
78     */
79    @EventListener(ApplicationReadyEvent.class)
80    public void init() {
81      log.info("""
82  
83              *********************************************************************************
84              * {}
85              *********************************************************************************
86              * apiPaths = {}
87              *********************************************************************************""",
88          ClassUtils.getUserClass(getClass()).getSimpleName(),
89          properties.getApiPaths());
90    }
91  
92    /**
93     * Builds api exception handler bean.
94     *
95     * @param errorAttributes the error attributes
96     * @param webProperties the web properties
97     * @param applicationContext the application context
98     * @param serverCodecConfigurer the server codec configurer
99     * @param restApiExceptionMapper the rest api exception mapper
100    * @return the api exception handler bean
101    */
102   @Bean
103   @Order(-2) // to have a higher priority than DefaultErrorWebExceptionHandler
104   public ApiExceptionHandler apiExceptionHandler(
105       ObjectProvider<ErrorAttributes> errorAttributes,
106       ObjectProvider<WebProperties> webProperties,
107       ApplicationContext applicationContext,
108       ObjectProvider<ServerCodecConfigurer> serverCodecConfigurer,
109       ObjectProvider<RestApiExceptionMapper> restApiExceptionMapper) {
110 
111     Assert.notNull(
112         errorAttributes.getIfAvailable(),
113         "Error attributes must be present.");
114     Assert.notNull(
115         serverCodecConfigurer.getIfAvailable(),
116         "Server codec configurer must be present.");
117     Assert.notNull(
118         restApiExceptionMapper.getIfAvailable(),
119         "Rest api exception mapper must be present.");
120 
121     return new ApiExceptionHandler(
122         properties.getApiPaths(),
123         errorAttributes.getIfAvailable(),
124         webProperties.getIfAvailable(WebProperties::new).getResources(),
125         applicationContext,
126         serverCodecConfigurer.getIfAvailable(),
127         restApiExceptionMapper.getIfAvailable());
128   }
129 
130 }
