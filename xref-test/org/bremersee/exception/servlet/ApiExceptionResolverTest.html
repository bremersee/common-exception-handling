87    /**
88     * Resolve exception with accept json.
89     *
90     * @param softly the softly
91     */
92    @Test
93    void resolveExceptionWithAcceptJson(SoftAssertions softly) {
94      HttpServletRequest request = mock(HttpServletRequest.class);
95      when(request.getServletPath()).thenReturn("/api/resource");
96      when(request.getRequestURI()).thenReturn("/api/resource");
97      when(request.getHeaderNames())
98          .thenReturn(Collections.enumeration(List.of(HttpHeaders.ACCEPT)));
99      when(request.getHeaders(eq(HttpHeaders.ACCEPT)))
100         .thenReturn(Collections.enumeration(List.of(MediaType.APPLICATION_JSON_VALUE)));
101 
102     when(request.getAttribute(eq(WebUtils.INCLUDE_REQUEST_URI_ATTRIBUTE)))
103         .thenReturn(true);
104 
105     RestApiException expected = RestApiException.builder()
106         .status(404)
107         .error(HttpStatus.NOT_FOUND.getReasonPhrase())
108         .build();
109 
110     when(exceptionMapper.build(any(Throwable.class), anyString(), any()))
111         .thenReturn(expected);
112 
113     HttpServletResponse response = mock(HttpServletResponse.class);
114 
115     ServiceException exception = mock(ServiceException.class);
116 
117     ModelAndView actual = target.resolveException(request, response, null, exception);
118 
119     softly.assertThat(actual)
120         .isNotNull()
121         .extracting(ModelAndView::getStatus)
122         .isEqualTo(HttpStatus.NOT_FOUND);
123     softly.assertThat(actual)
124         .isNotNull()
125         .extracting(ModelAndView::getModel)
126         .extracting(model -> model.get(MODEL_KEY))
127         .isEqualTo(expected);
128   }
